async function fetchData() { try { // ДОДАЄМО ПАРОЛЬ У ЗАПИТ (Authorization) const response = await fetch('https://monitor.proptech.com.ua', { method: 'GET', headers: { 'Authorization': 'Kyiv-2026', // Твій секретний токен 'Content-Type': 'application/json' } }); if (!response.ok) { throw new Error(`Помилка доступу: ${response.status}`); } const raw = await response.json(); // Якщо база видала один об'єкт, робимо з нього масив const sensors = Array.isArray(raw) ? raw : [raw]; const grid = document.getElementById('sensor-grid'); const countLabel = document.getElementById('active-count'); grid.innerHTML = ''; countLabel.textContent = `Nodes Online: ${sensors.length}`; sensors.forEach(s => { if(!s.location) return; const isAlarm = s.moisture > 60; const card = document.createElement('div'); card.className = 'sensor-card p-8 rounded-[2rem] flex flex-col justify-between'; card.innerHTML = `
${s.location}
${s.moisture}%
Relative Humidity
${s.battery || 95}% ${new Date(s.updated || Date.now()).toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'})}
`; grid.appendChild(card); }); } catch (err) { console.error("Connection failed:", err); const grid = document.getElementById('sensor-grid'); grid.innerHTML = `
Помилка авторизації або доступу
`; } } // Оновлюємо кожні 30 сек setInterval(fetchData, 30000); fetchData();
